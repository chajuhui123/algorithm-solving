function solution(word) {
  const words = ["A", "E", "I", "O", "U"];
  const plus = [781, 156, 31, 6, 1];
  return word
    .split("")
    .reduce((acc, ch, idx) => acc + words.indexOf(ch) * plus[idx] + 1, 0);
}
